━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   <meta name="google-site-verification" content="ABC123XYZ..." />

2. Copia SOLO: ABC123XYZ...

3. Edita /index.html línea 77:
   Cambia:

4. GUARDA y PUBLICA de nuevo desde Figma

5. Vuelve a Google Search Console → "Verificar"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. En Google Search Console → "Sitemaps"
2. Escribe: sitemap.xml
3. Clic en "Enviar"

2. Pega: https://wearehersafe.com/
3. Espera inspección → "Solicitar indexación"
